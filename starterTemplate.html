animations.css 
/* Bounce Animation */
@keyframes bounce {
    0%, 100% { transform: translateY(0); }
    50% { transform: translateY(-30px); }
}

/* Pulse Animation */
@keyframes pulse {
    0% { 
        transform: scale(1); 
        opacity: 1; 
    }
    50% { 
        transform: scale(1.1); 
        opacity: 0.7; 
    }
    100% { 
        transform: scale(1); 
        opacity: 1; 
    }
}

/* Spin Animation */
@keyframes spin {
    0% { transform: rotate(0deg); }
    100% { transform: rotate(360deg); }
}

/* Fade Animation */
@keyframes fadeInOut {
    0%, 100% { opacity: 0.3; }
    50% { opacity: 1; }
}

/* Usage Examples */
.bounce-element {
    animation: bounce 2s infinite;
}

.pulse-element {
    animation: pulse 2s infinite;
}

.spin-element {
    animation: spin 2s linear infinite;
}

.fade-element {
    animation: fadeInOut 3s infinite;
}
